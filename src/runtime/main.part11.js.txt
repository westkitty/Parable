    const d = distance2(shrine.x, shrine.z, x, z);
    if (d < bestDistance) {
      best = shrine;
      bestDistance = d;
    }
  }
  return best;
}

function frightenVillagers(point, radius, amount) {
  for (const villager of world.villagers) {
    if (distance2(villager.x, villager.z, point.x, point.z) <= radius) {
      villager.fear = clamp01(villager.fear + amount);
      villager.timer = 0;
    }
  }
}

function inspireVillagers(point, radius, amount) {
  for (const villager of world.villagers) {
    if (distance2(villager.x, villager.z, point.x, point.z) <= radius) {
      villager.faith = clamp01(villager.faith + amount);
      villager.fear = clamp01(villager.fear - amount * 0.5);
    }
  }
}

function shelterVillagers(point, radius) {
  for (const villager of world.villagers) {
    if (distance2(villager.x, villager.z, point.x, point.z) <= radius) {
      villager.state = "shelter";
      villager.targetX = point.x + (world.rng() - 0.5) * 3.5;
      villager.targetZ = point.z + (world.rng() - 0.5) * 3.5;
      villager.fear = clamp01(villager.fear - 0.25);
      villager.timer = 3;
    }
  }
}

function pushVillagers(point, radius, strength) {
  for (const villager of world.villagers) {
    const d = distance2(villager.x, villager.z, point.x, point.z);
    if (d <= radius) {
      const away = new THREE.Vector2(villager.x - point.x, villager.z - point.z).normalize();
      villager.x += away.x * strength * 2.5;
      villager.z += away.y * strength * 2.5;
      villager.timer = 0;
    }
  }
}

function simplifyPoints(points) {
  if (points.length <= 2) return points;
  const simplified = [points[0]];
  for (let i = 1; i < points.length - 1; i += 1) {
    const previous = simplified[simplified.length - 1];
    const current = points[i];
    if (Math.hypot(current.x - previous.x, current.y - previous.y) >= 3) simplified.push(current);
  }
  simplified.push(points.at(-1));
  return simplified;
}

function interpolateScreenPoints(a, b, maxStep) {
  const dx = b.x - a.x;
  const dy = b.y - a.y;
  const steps = Math.max(1, Math.ceil(Math.hypot(dx, dy) / maxStep));
  const output = [];
  for (let i = 1; i <= steps; i += 1) {
    const t = i / steps;
    output.push({ x: a.x + dx * t, y: a.y + dy * t, time: b.time });
  }
  return output;
}
