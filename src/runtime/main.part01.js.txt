function initStats() {
  statsHost.innerHTML = "";

  for (const stat of statDefs) {
    const row = document.createElement("div");
    row.className = "stat-row";
    row.dataset.key = stat.key;

    const title = document.createElement("div");
    title.className = "stat-title";
    title.textContent = stat.label;

    const value = document.createElement("div");
    value.className = "stat-value";

    const detail = document.createElement("div");
    detail.className = "stat-detail";

    row.append(title, value, detail);
    statsHost.append(row);
  }
}

function initMiracleUI() {
  miracleListHost.innerHTML = "";

  for (const miracle of MIRACLES) {
    const button = document.createElement("button");
    button.type = "button";
    button.className = "miracle-card";
    button.dataset.id = miracle.id;
    button.addEventListener("click", () => {
      selectMiracle(miracle.id);
    });

    miracleButtons.set(miracle.id, button);
    miracleListHost.append(button);
  }

  updateMiracleUI();
}

function createWorld(seed) {
  const rng = mulberry32(seed);

  const shrines = Array.from({ length: SHRINE_COUNT }, (_, index) => {
    const angle = (TAU / SHRINE_COUNT) * index + rng() * 0.5;
    const distance = 210 + rng() * 220;
    return {
      id: index,
      x: Math.cos(angle) * distance,
      y: Math.sin(angle) * distance * 0.76,
      aura: 34 + rng() * 16,
      size: 0.8 + rng() * 0.4,
    };
  });

  const villages = Array.from({ length: 4 }, (_, index) => {

    const distance = 160 + rng() * 220;
    return {
      id: index,
      x: Math.cos(angle) * distance,
      y: Math.sin(angle) * distance * 0.75,
      hearth: 0.52 + rng() * 0.28,
