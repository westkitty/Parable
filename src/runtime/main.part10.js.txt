  for (const villager of world.villagers) {
    const view = villager.view;
    if (!view) continue;
    view.position.set(villager.x, 0.25, villager.z);
    view.rotation.y += dt * (villager.state === "pray" ? 0.4 : 1.2);
    view.userData.hood.material.color.copy(villagerColor(villager.state));
    const bob = Math.sin(visualClock * 6 + villager.id) * 0.05;
    view.position.y = 0.25 + bob;
  }

  for (const golem of world.golems) {
    if (!golem.view) continue;
    golem.view.position.set(golem.x, 0.35 + Math.sin(visualClock * 2 + golem.id) * 0.04, golem.z);
    golem.view.userData.core.rotation.z += dt * 1.8;
  }
}

function updateEffects(dt) {
  for (let i = effectGroup.children.length - 1; i >= 0; i -= 1) {
    const object = effectGroup.children[i];
    const effect = object.userData.effect;
    if (!effect) continue;
    effect.age += dt;
    const t = Math.min(1, effect.age / effect.duration);
    object.traverse((child) => {
      if (child.material && "opacity" in child.material) child.material.opacity *= Math.max(0.02, 1 - t * 0.12);
    });
    if (effect.type === "fade-scale") object.scale.setScalar(1 + t * 0.7);
    if (effect.type === "fade-spin") object.rotation.y += dt * 2.2;
    if (effect.age >= effect.duration) effectGroup.remove(object);
  }
}

function updateTerrainColorsIfNeeded() {
  if (!terrainColorDirty) return;
  terrainColorDirty = false;
  const colorAttr = terrainMesh.geometry.attributes.color;
  const positions = terrainMesh.geometry.attributes.position;
  const temp = new THREE.Color();
  for (let i = 0; i < positions.count; i += 1) {
    const x = positions.getX(i);
    const z = positions.getZ(i);
    const radial = Math.sqrt(x * x + z * z) / WORLD_RADIUS;
    if (radial > 1) {
      temp.set("#0a1a22");
    } else {
      const region = nearestRegion(x, z);
      temp.copy(COLORS.terrain);
      temp.lerp(COLORS.fertile, region.fertility * 0.34);
      temp.lerp(COLORS.scorch, region.scorch * 0.58);
      temp.lerp(COLORS.rival, region.corruption * 0.46 + region.rival * 0.22);
      temp.lerp(COLORS.sacred, region.sacred * 0.28 + region.sanctity * 0.18);
      temp.lerp(COLORS.stone, region.stone * 0.24);
      temp.lerp(COLORS.water, region.wet * 0.2);
    }
    colorAttr.setXYZ(i, temp.r, temp.g, temp.b);
  }
  colorAttr.needsUpdate = true;
}

function updateStats() {
  const owners = countOwners();
  const states = countVillagerStates();
  const values = {
    worship: [Math.floor(world.worship), "Spend worship through spiral-armed miracles."],
    sacredRegions: [owners.sacred, "Long-lived regions currently leaning sacred."],

    villagerState: [world.villagers.length, stateSummary(states)],

  };
  for (const [key, [value, detail]] of Object.entries(values)) {
    const row = statsHost.querySelector(`[data-key="${key}"]`);
    if (!row) continue;
    row.querySelector(".stat-value").textContent = value;
    row.querySelector(".stat-detail").textContent = detail;
  }
}

function updateRegionOwnership() {
  for (const region of world.regions) {
    const sacredScore = region.sacred + region.sanctity + region.shelter * 0.25;
    const rivalScore = region.rival + region.corruption * 0.7;
    const previous = region.owner;
    if (sacredScore > rivalScore + 0.2) region.owner = "sacred";
    else if (rivalScore > sacredScore + 0.18) region.owner = "rival";
    else region.owner = "contested";
    if (previous !== region.owner) terrainColorDirty = true;
  }
}

function resizeRenderer() {
  const rect = canvas.getBoundingClientRect();
  const width = Math.max(1, Math.floor(rect.width));
  const height = Math.max(1, Math.floor(rect.height));
  renderer.setSize(width, height, false);
  camera.aspect = width / height;
  camera.updateProjectionMatrix();
}

function regionsWithin(x, z, radius) {
  return world.regions.filter((region) => distance2(region.x, region.z, x, z) <= radius);
}

function nearestRegion(x, z) {
  let best = world.regions[0];
  let bestDistance = Infinity;
  for (const region of world.regions) {
    const d = distance2(region.x, region.z, x, z);
    if (d < bestDistance) {
      best = region;
      bestDistance = d;
    }
  }
  return best;
}
