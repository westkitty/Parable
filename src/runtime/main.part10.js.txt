  return height;
}

function getBaseCorruptionAt(x, y) {
  const distanceToRival = distance(x, y, world.rivalCore.x, world.rivalCore.y);
  return (1 - smoothstep(world.rivalCore.radius * 0.9, world.rivalCore.radius * 2.4, distanceToRival)) * world.rivalCore.corruption;
}

function weakenZonesNear(x, y, radius, amount, markDirty = true) {
  let affected = 0;

  for (const zone of world.zones) {
    if (!HARMFUL_ZONES.has(zone.kind)) {
      continue;
    }

    const d = distance(x, y, zone.x, zone.y);
    if (d > radius + zone.radius * 0.25) {
      continue;
    }

    const strength = Math.max(0.08, 1 - d / (radius + zone.radius));
    zone.power = Math.max(0, zone.power - amount * strength);
    zone.duration = Math.max(zone.age + 0.25, zone.duration - amount * 8 * strength);
    affected += 1;

    if (markDirty && zoneAffectsTerrain(zone)) {
      world.terrainDirty = true;
    }
  }

  return affected;
}

function forEachVillagerInRadius(x, y, radius, callback) {
  let count = 0;
  for (const villager of world.villagers) {
    if (distance(x, y, villager.x, villager.y) <= radius) {
      callback(villager);
      count += 1;
    }
  }
  return count;
}

function countZones(kind) {
  return world.zones.filter((zone) => zone.kind === kind).length;
}

function zoneAffectsTerrain(zone) {
  return Boolean(zone.terrainLift || zone.terrainCrater || zone.fertility || zone.sanctity || zone.corruption || zone.scorch);
}

function isPointInsideIsland(x, y) {
  return Math.hypot(x / 1.02, y / 0.82) <= WORLD_RADIUS + 60;
}

function getMiracleColorForEffect(kind) {
  const miracle = MIRACLES.find((entry) => entry.id === kind);
  if (miracle) {
    return miracle.color;
  }

  const fallback = {
    blight: 0xb24646,
    purgeGlow: 0xe8efe8,
    stoneShape: 0xbebaa8,
    ritualArm: 0xf4c768,
  };

  return fallback[kind] ?? 0xf4c768;
}

function disposeObject(object) {
  object.traverse((child) => {
    if (child.geometry) {
      child.geometry.dispose();
    }

    if (child.material) {
      if (Array.isArray(child.material)) {
        child.material.forEach((material) => material.dispose());
      } else {
        child.material.dispose();
      }
    }
  });
}

function distance(ax, ay, bx, by) {
  return Math.hypot(ax - bx, ay - by);
}

function clamp(value, min, max) {
  return Math.max(min, Math.min(max, value));
}

function smoothstep(edge0, edge1, x) {
  const t = clamp((x - edge0) / (edge1 - edge0), 0, 1);
  return t * t * (3 - 2 * t);
}

function damp(current, target, lambda, dt) {
  return THREE.MathUtils.lerp(current, target, 1 - Math.exp(-lambda * dt));
}

function mulberry32(seed) {
  let t = seed;
  return function next() {
    t += 0x6d2b79f5;
    let r = Math.imul(t ^ (t >>> 15), t | 1);
    r ^= r + Math.imul(r ^ (r >>> 7), r | 61);
    return ((r ^ (r >>> 14)) >>> 0) / 4294967296;
  };
}
