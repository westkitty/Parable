    const worldPoint = getWorldPoint(event);
    if (worldPoint && worldPoint.length() <= WORLD_RADIUS + 2) {
      pointer.worldPoints.push(worldPoint.clone());
      pointer.lastWorld.copy(worldPoint);
      createGestureTrace(pointer.worldPoints.slice(-18), ritual.armed ? "#ff875b" : "#f4c768", 0.42);
    }
  });

  canvas.addEventListener("pointerup", finishPointerGesture);
  canvas.addEventListener("pointercancel", finishPointerGesture);
}

function finishPointerGesture(event) {
  if (!pointer.active || pointer.id !== event.pointerId) return;
  pointer.active = false;
  const now = performance.now();
  const screenPoints = simplifyPoints(pointer.points);
  const worldAnchor = pointer.worldPoints.length ? pointer.worldPoints.at(-1).clone() : pointer.lastWorld.clone();
  const analysis = analyzeGesture(screenPoints);
  handleGesture(analysis, worldAnchor, now);
  pointer.points = [];
  pointer.worldPoints = [];
  pointer.id = null;
}

function screenPointFromEvent(event) {
  const rect = canvas.getBoundingClientRect();
  return { x: event.clientX - rect.left, y: event.clientY - rect.top, time: performance.now() };
}

function getWorldPoint(event) {
  const rect = canvas.getBoundingClientRect();
  pointerNdc.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
  pointerNdc.y = -(((event.clientY - rect.top) / rect.height) * 2 - 1);
  raycaster.setFromCamera(pointerNdc, camera);
  const target = new THREE.Vector3();
  const hit = raycaster.ray.intersectPlane(groundPlane, target);
  return hit ? target : null;
}

function handleGesture(analysis, worldPoint, now) {
  gestureLabel.textContent = analysis.label;
  gestureDetail.textContent = analysis.detail;

  if (ritual.armed && now > ritual.expiresAt) {
    ritual.armed = false;
    ritual.anchor = null;
    updateRitualHud("Expired", "The armed spiral went cold. Draw another clockwise spiral.");
  }

  if (!ritual.armed) {
    if (analysis.kind === "clockwise-spiral") {
      armRitual(worldPoint, now, analysis);
      return;
    }
    updateRitualHud("Unarmed", "That was not a clockwise spiral. The ritual did not arm.");
    return;
  }

  const miracle = miracleById.get(analysis.miracleId || "");
  if (!miracle) {
    updateRitualHud("Armed", "The spiral is armed, but that finisher is not recognized.");
    return;
  }

  const castPoint = ritual.anchor || worldPoint;
  const result = castMiracle(miracle, castPoint, now);
  if (result.ok) {
    ritual.armed = false;
    ritual.anchor = null;
    ritual.expiresAt = 0;
    updateRitualHud("Unarmed", `${miracle.name} completed. Draw a new clockwise spiral.`);
  } else {
    updateRitualHud("Armed", result.reason);
  }
}

function armRitual(worldPoint, now, analysis) {
  ritual.armed = true;
  ritual.anchor = worldPoint.clone();
  ritual.expiresAt = now + ARM_TIMEOUT;
  createRingEffect(worldPoint, 2.2, "#f4c768", 1.2, 1.2);
  createColumnEffect(worldPoint, "#f4c768", 1.2);
  updateRitualHud("Armed", "Clockwise spiral accepted. Draw a finisher before the glow dies.");
  logEvent(`Ritual armed by ${analysis.label}.`);
}

function analyzeGesture(points) {
  if (points.length < 6) {
    return { kind: "too-short", label: "Too short", detail: "Gesture needs a longer stroke." };
  }
  const sampled = resamplePoints(points, 42);
  const box = boundingBox(sampled);
  const width = Math.max(1, box.maxX - box.minX);
  const height = Math.max(1, box.maxY - box.minY);
  const totalLength = pathLength(sampled);
  const direct = Math.hypot(sampled.at(-1).x - sampled[0].x, sampled.at(-1).y - sampled[0].y);
  const aspect = width / height;
  const turns = countHorizontalTurns(sampled);
  const verticalTravel = sampled.at(-1).y - sampled[0].y;
  const horizontalTravel = sampled.at(-1).x - sampled[0].x;
  const rotation = totalSignedRotation(sampled);
  const closure = Math.hypot(sampled.at(-1).x - sampled[0].x, sampled.at(-1).y - sampled[0].y) / Math.max(width, height);

  if (rotation > TAU * 1.05 && totalLength > Math.max(width, height) * 4.2 && closure < 0.85) {
    return { kind: "clockwise-spiral", label: "Clockwise spiral", detail: "Ritual armed. This is the only accepted opener." };
  }

  if (height > width * 1.25 && verticalTravel > height * 0.25 && turns >= 3 && totalLength > height * 1.65) {
    return { kind: "finisher", miracleId: "fireball", label: "Vertical zigzag", detail: "Fireball finisher recognized." };
  }

  if (height > width * 1.9 && verticalTravel > height * 0.55 && turns <= 1) {
    return { kind: "finisher", miracleId: "rain", label: "Downward line", detail: "Rain finisher recognized." };
  }

  if (closure < 0.32 && Math.abs(width - height) / Math.max(width, height) < 0.42 && totalLength > Math.max(width, height) * 2.4) {
    return { kind: "finisher", miracleId: "blessing", label: "Closed circle", detail: "Blessing finisher recognized." };
  }

  if (turns >= 1 && width > height * 0.72 && height > width * 0.45 && hasCrossingStroke(sampled)) {
    return { kind: "finisher", miracleId: "purge", label: "Cross stroke", detail: "Purge finisher recognized." };
  }

  if (verticalTravel < -height * 0.25 && turns >= 2 && height > width * 0.75) {
    return { kind: "finisher", miracleId: "fertility", label: "Upward sprout", detail: "Fertility finisher recognized." };
  }

  if (closure < 0.48 && aspect > 0.65 && aspect < 1.55 && countCorners(sampled) >= 3) {
    return { kind: "finisher", miracleId: "stone-shape", label: "Square seal", detail: "Stone Shape finisher recognized." };
