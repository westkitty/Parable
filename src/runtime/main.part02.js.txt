function initUi() {
  statsHost.innerHTML = "";
  for (const stat of statDefs) {
    const row = document.createElement("div");
    row.className = "stat-row";
    row.dataset.key = stat.key;
    row.innerHTML = `<div class="stat-title"></div><div class="stat-value"></div><div class="stat-detail"></div>`;
    row.querySelector(".stat-title").textContent = stat.label;
    statsHost.append(row);
  }

  miracleHost.innerHTML = "";
  for (const miracle of MIRACLES) {
    const card = document.createElement("article");
    card.className = "miracle-card";
    card.innerHTML = `
      <header><strong></strong><code></code></header>
      <p></p>
      <div class="miracle-meta"><span></span><span></span></div>
    `;
    card.querySelector("strong").textContent = miracle.name;
    card.querySelector("code").textContent = miracle.finisher;
    card.querySelector("p").textContent = miracle.purpose;
    const [cost, cooldown] = card.querySelectorAll(".miracle-meta span");
    cost.textContent = `Cost ${miracle.cost}`;
    cooldown.textContent = `Cooldown ${miracle.cooldown}s`;
    miracleHost.append(card);
  }

  logEvent("Draw a clockwise spiral to arm the ritual. Follow it with a finishing glyph.");
  updateRitualHud("Unarmed", "Draw a clockwise spiral on the island.");
}

  const villages = createVillages(rng, regions);
  const shrines = createShrines(rng, regions);
  const villagers = Array.from({ length: VILLAGER_COUNT }, (_, index) => createVillager(index, villages[index % villages.length], rng));

  for (const region of regions) {
    const rivalDistance = distance2(region.x, region.z, rival.x, rival.z);
    region.rival += Math.max(0, 1 - rivalDistance / 32) * 0.42;
    region.corruption += Math.max(0, 1 - rivalDistance / 24) * 0.3;
  }

    time: 0,

    influence: 0.18,
