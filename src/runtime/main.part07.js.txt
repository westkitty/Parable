  if (isCaret(sampled, box)) {
    return { kind: "finisher", miracleId: "beacon", label: "Caret peak", detail: "Beacon finisher recognized." };
  }

  if (aspect > 1.65 && turns >= 2) {
    return { kind: "finisher", miracleId: "wind-lash", label: "Horizontal wave", detail: "Wind Lash finisher recognized." };
  }

  if (isHook(sampled, box)) {
    return { kind: "finisher", miracleId: "golem-wake", label: "Hooked stroke", detail: "Golem Wake finisher recognized." };
  }

  if (isRoofWard(sampled, box)) {
    return { kind: "finisher", miracleId: "sanctuary", label: "Roof ward", detail: "Sanctuary finisher recognized." };
  }

  return { kind: "unknown", label: "Unknown glyph", detail: "No miracle matched. Use the spellbook glyph names as a guide." };
}

function castMiracle(miracle, point, now) {
  const secondsNow = now / 1000;
  if (world.worship < miracle.cost) return { ok: false, reason: `${miracle.name} needs ${miracle.cost} worship.` };
  if (secondsNow < cooldowns[miracle.id]) {
    const remain = Math.ceil(cooldowns[miracle.id] - secondsNow);
    return { ok: false, reason: `${miracle.name} is cooling down for ${remain}s.` };
  }
  world.worship -= miracle.cost;
  cooldowns[miracle.id] = secondsNow + miracle.cooldown;
  applyMiracle(miracle, point);
  createMiracleEffect(miracle, point);
  terrainColorDirty = true;
  logEvent(`${miracle.name} cast at the spiral anchor.`);
  return { ok: true };
}

function applyMiracle(miracle, point) {
  const affected = regionsWithin(point.x, point.z, miracle.radius);
  for (const region of affected) {
    const strength = Math.max(0.1, 1 - distance2(region.x, region.z, point.x, point.z) / miracle.radius);
    switch (miracle.id) {
      case "fireball":
        region.scorch = clamp01(region.scorch + 0.55 * strength);
        region.rival = clamp01(region.rival - 0.22 * strength);
        region.corruption = clamp01(region.corruption - 0.18 * strength);
        region.fertility = clamp01(region.fertility - 0.2 * strength);
        world.fear = clamp01(world.fear + 0.08 * strength);
        frightenVillagers(point, miracle.radius + 3, 0.7);
        break;
      case "rain":
        region.wet = clamp01(region.wet + 0.65 * strength);
        region.scorch = clamp01(region.scorch - 0.35 * strength);
        region.fertility = clamp01(region.fertility + 0.22 * strength);
        region.food = clamp01(region.food + 0.22 * strength);
        region.corruption = clamp01(region.corruption - 0.12 * strength);
        world.food += 3 * strength;
        break;
      case "blessing":
        region.sacred = clamp01(region.sacred + 0.3 * strength);
        region.sanctity = clamp01(region.sanctity + 0.28 * strength);
        inspireVillagers(point, miracle.radius, 0.22 * strength);
        break;
      case "purge":
        region.rival = clamp01(region.rival - 0.42 * strength);
        region.corruption = clamp01(region.corruption - 0.5 * strength);
        region.sacred = clamp01(region.sacred + 0.12 * strength);
        break;
      case "fertility":
        region.fertility = clamp01(region.fertility + 0.48 * strength);
        region.food = clamp01(region.food + 0.4 * strength);
        region.scorch = clamp01(region.scorch - 0.14 * strength);
        world.food += 4.5 * strength;
        break;
      case "stone-shape":
        region.stone = clamp01(region.stone + 0.48 * strength);
        region.shelter = clamp01(region.shelter + 0.38 * strength);
        region.scorch = clamp01(region.scorch - 0.05 * strength);
        addStoneRidge(point, strength);
        break;
      case "beacon":
        region.sacred = clamp01(region.sacred + 0.38 * strength);
        region.sanctity = clamp01(region.sanctity + 0.22 * strength);
        region.rival = clamp01(region.rival - 0.08 * strength);
        addBeacon(point);
        break;
      case "wind-lash":
        region.rival = clamp01(region.rival - 0.18 * strength);
        region.corruption = clamp01(region.corruption - 0.12 * strength);
        pushVillagers(point, miracle.radius, strength);
        break;
      case "golem-wake":
        region.stone = clamp01(region.stone + 0.16 * strength);
        wakeGolem(point);
        break;
      case "sanctuary":
        region.shelter = clamp01(region.shelter + 0.58 * strength);
        region.sanctity = clamp01(region.sanctity + 0.34 * strength);
        region.rival = clamp01(region.rival - 0.2 * strength);
        region.corruption = clamp01(region.corruption - 0.15 * strength);
        world.fear = clamp01(world.fear - 0.15 * strength);
        shelterVillagers(point, miracle.radius);
        break;
      default:
        break;
    }
  }
  updateRegionOwnership();
}

function createMiracleEffect(miracle, point) {
  const color = {
    fireball: "#ff875b",
    rain: "#5aaec1",
    blessing: "#f4c768",
    purge: "#edf6e2",
    fertility: "#6fcf71",
    "stone-shape": "#9da8a2",
    beacon: "#ffe49a",
    "wind-lash": "#a8f0e7",
    "golem-wake": "#6bc8be",
    sanctuary: "#6bc8be",
  }[miracle.id] || "#f4c768";
  createRingEffect(point, miracle.radius, color, 1.15, 1.4);
  createColumnEffect(point, color, 1.3);
  if (miracle.id === "rain") createRainEffect(point, miracle.radius);
  if (miracle.id === "fireball") createFireballImpact(point);
  if (miracle.id === "sanctuary") createDomeEffect(point, miracle.radius);
}

  const mesh = new THREE.Mesh(

    new THREE.MeshBasicMaterial({ color, transparent: true, opacity: 0.85 })
